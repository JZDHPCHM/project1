SELECT
   '续期收费',
   T1.FD_NAME AS 机构,
   CASE WHEN T.CHANNEL = 'BK' THEN '银保'
       WHEN T.CHANNEL = 'GP' THEN '团险'
       WHEN T.CHANNEL = 'CS' THEN '客服'
       WHEN T.CHANNEL = 'FC' THEN '个险'
       WHEN T.CHANNEL = 'AD' THEN '多元'
       WHEN T.CHANNEL = 'RP' THEN '续期' ELSE T.CHANNEL END AS 渠道
  ,T.RECEIPTS_AMT   AS 实收保费
  ,T.RECEIVABLE_AMT AS 应收保费
  ,T.RESULT_RATE    AS 比率
  ,T.SCORE          AS 得分
FROM E_ADS_CONTINUE_CHARGE_RATE T
LEFT JOIN FR_AA_ORG T1
ON T.SEC_ORG = T1.FD_CODE
ORDER BY T.SEC_ORG;

SELECT
   '新契约回访',
   T1.FD_NAME AS 机构,
   CASE WHEN T.CHANNEL = 'BK' THEN '银保'
       WHEN T.CHANNEL = 'GP' THEN '团险'
       WHEN T.CHANNEL = 'CS' THEN '客服'
       WHEN T.CHANNEL = 'FC' THEN '个险'
       WHEN T.CHANNEL = 'AD' THEN '多元'
       WHEN T.CHANNEL = 'RP' THEN '续期' ELSE T.CHANNEL END AS 渠道
  ,T.VISIT_NUM AS 回访件数
  ,T.POLICY_NUM AS 保单件数
  ,T.RESULT_RATE AS 比率
  ,T.SCORE       AS 得分
FROM E_ADS_CONTRACT_VISIT_RATE T
LEFT JOIN FR_AA_ORG T1
ON T.ORG_NAME = T1.FD_CODE
ORDER BY T.ORG_NAME;

       WHEN T.CHANNEL = 'GP' THEN '团险'
       WHEN T.CHANNEL = 'CS' THEN '客服'
